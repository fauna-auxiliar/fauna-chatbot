from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/chat")
async def chat_endpoint(payload: dict):
    message = payload.get("message")
    # Aquí iría tu integración con OpenAI
    return {"response": f"Recibí tu mensaje: {message}"}
